#include "s21_decimal.h"

int s21_truncate(s21_decimal value, s21_decimal *result) {
  for (int i = 0; i < 4; i++) {
    result->bits[i] = value.bits[i];
  }
  int scale = get_exp(value);
  result->bits[3] = value.bits[3] & 0x80000000;

while (scale) {
    result /= 10;
    scale--;
  }

  return 0;
}
